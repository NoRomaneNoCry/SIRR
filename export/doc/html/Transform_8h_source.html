66  Matrix4x4 mul( const Matrix4x4 &m1, const Matrix4x4 &m2 )
67  {
68  float r[4][4];
69 
70  for (int i = 0; i < 4; ++i)
71  for (int j = 0; j < 4; ++j)
72  r[i][j] =
73  m1.m[i][0] * m2.m[0][j] +
74  m1.m[i][1] * m2.m[1][j] +
75  m1.m[i][2] * m2.m[2][j] +
76  m1.m[i][3] * m2.m[3][j];
