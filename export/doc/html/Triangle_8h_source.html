100  bool Intersect( const Ray &ray, const float htmax,
101  float &rt, float &ru, float&rv ) const
102  {
103  /* begin calculating determinant - also used to calculate U parameter */
104  Vector ac(a, c);
105  const Vector pvec= Cross(ray.d, ac);
106 
107  /* if determinant is near zero, ray lies in plane of triangle */
108  Vector ab(a, b);
109  const float det= Dot(ab, pvec);
110  if (det > -EPSILON && det < EPSILON)
111  return false;
112 
113  const float inv_det= 1.0f / det;
114 
115  /* calculate distance from vert0 to ray origin */
116  const Vector tvec(a, ray.o);
117 
118  /* calculate U parameter and test bounds */
119  const float u= Dot(tvec, pvec) * inv_det;
120  if(u < 0.0f || u > 1.0f)
121  return false;
122 
123  /* prepare to test V parameter */
124  const Vector qvec= Cross(tvec, ab);
125 
126  /* calculate V parameter and test bounds */
127  const float v= Dot(ray.d, qvec) * inv_det;
128  if(v < 0.0f || u + v > 1.0f)
129  return false;
130 
131  /* calculate t, ray intersects triangle */
132  rt= Dot(ac, qvec) * inv_det;
133  ru= u;
134  rv= v;
135 
136  // ne renvoie vrai que si l'intersection est valide (comprise entre tmin et tmax du rayon)
137  return (rt < htmax && rt > RAY_EPSILON);
138  }
