204  void draw( )
205  {
206  if(m_buffer_id != -1u)
207  return; // must call end() before draw()
208 
209  // organise buffer data
210  unsigned int index_size= 0;
211  unsigned int data_size= 0;
212  for(unsigned int i= 0; i < m_buffers.size(); i++)
213  {
214  m_buffers[i].data_offset= data_size;
215  m_buffers[i].index_offset= index_size;
216 
217  index_size+= m_buffers[i].index.size();
218  data_size+= m_buffers[i].data.size();
219  }
220 
221  // update buffers
222  glBindVertexArray(0);
223 
224  // resize vertex buffer when necessary
230  GLint buffer_size;
231  glBindBuffer(GL_ARRAY_BUFFER, m_vertex_buffer);
232  glGetBufferParameteriv(GL_ARRAY_BUFFER, GL_BUFFER_SIZE, &buffer_size);
233  if((size_t) buffer_size < data_size * sizeof(vertex))
234  glBufferData(GL_ARRAY_BUFFER, data_size * sizeof(vertex), NULL, GL_STREAM_DRAW);
235 
236  // update vertex buffer
237  for(unsigned int i= 0; i < m_buffers.size(); i++)
238  glBufferSubData(GL_ARRAY_BUFFER,
239  m_buffers[i].data_offset * sizeof(vertex), m_buffers[i].data.size() * sizeof(vertex),
240  (const GLvoid *) &m_buffers[i].data.front());
241 
242  // resize index buffer when necessary
243  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, m_index_buffer);
244  glGetBufferParameteriv(GL_ELEMENT_ARRAY_BUFFER, GL_BUFFER_SIZE, &buffer_size);
245  if((size_t) buffer_size < index_size * sizeof(unsigned int))
246  glBufferData(GL_ELEMENT_ARRAY_BUFFER, index_size * sizeof(unsigned int), NULL, GL_STREAM_DRAW);
247 
248  // update index buffer
249  for(unsigned int i= 0; i < m_buffers.size(); i++)
250  glBufferSubData(GL_ELEMENT_ARRAY_BUFFER,
251  m_buffers[i].index_offset* sizeof(unsigned int), m_buffers[i].index.size() * sizeof(unsigned int),
252  (const GLvoid *) &m_buffers[i].index.front());
253 
254  // draw states
255  glBindVertexArray(m_vao);
256 
257  glPrimitiveRestartIndex(m_restart_index);
258  glEnable(GL_PRIMITIVE_RESTART);
259 
260  const void *program= NULL;
261  //~ printf("widgets batchs %lu\n", m_draws.size());
262  for(unsigned int i= 0; i < m_draws.size(); i++)
263  {
264  // bind shader
265  if(m_draws[i].program != program)
266  {
267  glUseProgram(m_draws[i].program->program);
268  // update projection matrix
269  assert(m_draws[i].program->projection > -1);
270  glUniformMatrix4fv(m_draws[i].program->projection, 1, GL_TRUE, (const GLfloat *) &m_projection);
271  program= m_draws[i].program;
272  }
273 
274  // set uniforms
275  m_draws[i].apply();
276 
277  // draw strips
278  glDrawElementsBaseVertex(GL_TRIANGLE_STRIP,
279  m_draws[i].end - m_draws[i].start,
280  GL_UNSIGNED_INT, (GLvoid *) ((m_buffers[i].index_offset + m_draws[i].start) * sizeof(unsigned int)),
281  m_buffers[i].data_offset);
282  }
283 
284  glUseProgram(0);
285  glBindVertexArray(0);
286  glBindBuffer(GL_ARRAY_BUFFER, 0);
287  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
288  glDisable(GL_PRIMITIVE_RESTART);
289  }
