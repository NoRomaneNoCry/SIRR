247  std::string readText( const std::string& filename )
248  {
249  std::string text;
250  FILE *in= fopen(filename.c_str(), "rb");
251  if(in == NULL)
252  return text;
253 
254  char last_char= 0;
255  char tmp[4096];
256  for(;;)
257  {
258  if(fgets(tmp, sizeof(tmp), in) == NULL)
259  break;
260 
261  // normalise les fins de lignes
262  for(int i= 0; tmp[i] != 0; i++)
263  {
264  char c= tmp[i];
265 
266  if((c == '\r') // windows + mac
267  || (c == '\n' && last_char != '\r')) // linux
268  text.push_back('\n');
269 
270  if(c != '\n') // les fins de lignes sont deja emises, ne pas dupliquer...
271  text.push_back(c);
272 
273  last_char= c;
274  }
275  }
276 
277  // termine la derniere ligne du fichier, si necessaire.
278  //~ if(text.back() != '\n') // c++11
279  if(text.empty() == false && text[text.size() -1] != '\n')
280  text.push_back('\n');
281 
282  fclose(in);
283  return text;
284  }
